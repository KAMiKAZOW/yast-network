DOC:
 ESCON = Enteprise System CONnector (something like a mixture of SCSI and
         Ethernet - its a channel, but can carry network traffic)
         - short: its a point-to point network interface

 IUCV  = Inter User Communication Vehicle (means: DMA based network pipe
         between Virtual Machine instances (like vmware) on an s390)
         - short: its a point-to point network interface

Detection() {
  CONFIG=$[];
  ReadConfig("eth");
  status(eth0) -> set up
  netprobe();
}

DetectionStop() {
  if set up -> stop(eth0)
  killall netprobe
  ReadDetection();
}
